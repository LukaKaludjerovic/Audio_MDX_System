1. Broj minuta slušanja audio snimka po polu, uzrastu, mestu korisnika koji je slušao audio snimak

    1.1. Zajedno po polu, uzrastu i mestu korisnika
        SELECT
            Measures.Trajanje ON COLUMNS,
            NON EMPTY
                [Korisnik.Polovi].Pol.Members *
                [Korisnik.Godista].Godiste.Members *
                [Korisnik.Mesta].Mesto.Members
            ON ROWS
        FROM Slusanja

    1.2. Odvojeno po polu korisnika
        SELECT
            Measures.Trajanje ON COLUMNS,
            NON EMPTY
                [Korisnik.Polovi].Pol.Members
            ON ROWS
        FROM Slusanja

    1.3. Odvojeno uzrastu korisnika
        SELECT
            Measures.Trajanje ON COLUMNS,
            NON EMPTY
                [Korisnik.Godista].Godiste.Members
            ON ROWS
        FROM Slusanja

    1.4. Odvojeno po mestu korisnika
        SELECT
            Measures.Trajanje ON COLUMNS,
            NON EMPTY
                [Korisnik.Mesta].Mesto.Members
            ON ROWS
        FROM Slusanja

2. Prosečnu ocenu za svaku kategoriju audio snimka, za svako mesto korisnika koji je vlasnik audio snimka

    2.1. Zajedno za svaku kategoriju i svako mesto vlasnika audio snimka
        SELECT
            Measures.[Prosecna ocena] ON COLUMNS,
            NON EMPTY
                [Kategorija.Nazivi].Naziv.Members *
                [Vlasnik.Mesta].Mesto.Members
            ON ROWS
        FROM Ocene

    2.2. Odvojeno po kategoriji audio snimka
        SELECT
            Measures.[Prosecna ocena] ON COLUMNS,
            NON EMPTY
                [Kategorija.Nazivi].Naziv.Members
            ON ROWS
        FROM Ocene

    2.3. Odvojeno po mestu vlasnika audio snimka
        SELECT
            Measures.[Prosecna ocena] ON COLUMNS,
            NON EMPTY
                [Vlasnik.Mesta].Mesto.Members
            ON ROWS
        FROM Ocene
